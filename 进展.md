#!/usr/bin/env bash
# Auto-configure WSL proxy through Windows Clash (port 7890).
# NAT mode: default gateway is usually the Windows host.
host_ip=$(ip route 2>/dev/null | awk '/default/{print $3; exit}')
if [ -z "$host_ip" ]; then
  host_ip=$(grep -m1 nameserver /etc/resolv.conf | tr -s " " | cut -d " " -f2)
fi

if [ -n "$host_ip" ]; then
  export http_proxy="http://$host_ip:7890"
  export https_proxy="http://$host_ip:7890"
  export HTTP_PROXY="$http_proxy"
  export HTTPS_PROXY="$https_proxy"
  export all_proxy="socks5://$host_ip:7890"
  export ALL_PROXY="$all_proxy"
  export no_proxy="localhost,127.0.0.1,::1,.local"
  export NO_PROXY="$no_proxy"
fi




daemon 失败点是 DNS 解析不了 host.docker.internal，而代理链路本身是通的。再补一眼 Docker 全局配置，确认没有其他代理项冲突。
GW=$(ip route | awk '/^default/{print $3; exit}')
echo "Use gateway: $GW"

sudo mkdir -p /etc/systemd/system/docker.service.d

sudo tee /etc/systemd/system/docker.service.d/http-proxy.conf >/dev/null <<EOF
[Service]
Environment="HTTP_PROXY=http://$GW:7890"
Environment="HTTPS_PROXY=http://$GW:7890"
Environment="NO_PROXY=localhost,127.0.0.1,::1,db,redis,backend,frontend,postgres"
EOF

sudo systemctl daemon-reload
sudo systemctl restart docker
